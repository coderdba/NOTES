INSTALL DOCKER
Reference - https://techviewleo.com/install-use-docker-ce-on-oracle-linux/

yum-config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo
yum install docker-ce docker-ce-cli containerd.io
systemctl enable docker
systemctl start docker

DOCKER TO BE RUN BY NON-ROOT USER
[root@localhost ~]# useradd myapp
[root@localhost ~]# id myapp
uid=1001(myapp) gid=1001(myapp) groups=1001(myapp)
[root@localhost ~]# usermod -aG docker myapp
[root@localhost ~]# id myapp
uid=1001(myapp) gid=1001(myapp) groups=1001(myapp),969(docker)
